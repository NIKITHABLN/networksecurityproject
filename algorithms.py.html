import hashlib

# initializing string
str = "Nikitha"

# encoding Nikitha using encode()
# then sending to SHA256()
result = hashlib.sha256(str.encode())

# printing the equivalent hexadecimal value.
print("The hexadecimal equivalent of SHA256 is : ")
print(result.hexdigest())

print("\r")
